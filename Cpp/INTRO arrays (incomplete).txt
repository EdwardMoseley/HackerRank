#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int n;
    scanf("%d", &n);
    int arr[n];
    int i = n-1;
    while(i >= 0){
        cin >> arr[i];
        i = i - 1;
    }
    i = 0;
    while (i <= n-1){
    cout << arr[i];
    cout << " ";
    i = i + 1;
    }
    return 0;
}
