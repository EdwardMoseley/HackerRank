#include <iostream>
#include <cstdio>
using namespace std;

int main(){
    int a;
    int b;
    cin >> a;
    cin >> b;
    
    string sDict[10] = {"one\n", "two\n", "three\n", "four\n", "five\n", "six\n", "seven\n", "eight\n", "nine\n"};
    
    for (int i = a; i < b; i++){
            cout << sDict[i-1];// << endl;
    }
    
    for (int j = a; j < b-1; j++){
        if (j % 2 == 0){
            cout << "even\n";
        } else {
            cout << "odd\n";
        }
    }
    
    return 0;
    
}
