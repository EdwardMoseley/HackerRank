import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        String str = sc.nextLine();
        int n = sc.nextInt();

        int[] count = new int[n];
        String[] store = new String[str.length()-n+1];

        //char[] store = new char[str.length()]; //let's check the first character first, then the whole string-- not sure what they're asking for precisely.

        System.out.println(str);
        System.out.println(n);
        
        for (int i = 0; i<=str.length()-n+1; i++){
            //store[i] = str.substring(i,i+n);
            if (str.substring(i, i+n).length() >= n){
                store[i] = str.substring(i,i+n);
            }
            //System.out.println(store[i]);
           //System.out.println(store[i].charAt(0));
           //Note: compareTo(String anotherString) 
           //Compares two strings lexicographically.
        }
            //Store array and THEN sort, may not be optimal? Research later.
        int j;
        boolean flag = true;   // set flag to true to begin first pass
        String tmp;    //holding variable
        while (flag){
        	System.out.println("while loop init");
                //flag = false;    //set flag to false awaiting a possible swap
                for (j=0;  j < store.length-1; j++){
                	System.out.println("for loop init");
                	if (store[j].compareTo(store[j+1]) >= 0){
                        	System.out.println("HELLO SWAP");
                        	tmp = store[j];                //swap elements
                                store[j] = store[j+1];
                                store[j+1] = tmp;
                                flag = true;              //shows swap 
                   	} else {
                        	System.out.println("break out");
                                flag = false;
			} 
		} 
	} 
      System.out.println(Arrays.toString(store));
	}
}
