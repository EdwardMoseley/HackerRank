f <- file("stdin")#Read in input

dat <- readLines(f)

n <- as.numeric(dat[1])

dat <- strsplit(dat[2:length(dat)], '')

grid <- matrix(rep(0, each = n*n), n, n)

for (i in 1:n){
    for (j in 1:n){
        #print("Inc...")
        if (as.character(dat[[i]][j]) == "p"){
            grid[i,j] <- grid[i,j] + n
        } else if (as.character(dat[[i]][j]) == "m"){
            grid[i,j] <- -1
        }
    }
}



grid
